Multiclass classification Problem Prodict B or NOT B

letters=read.csv('letters_ABPR.csv')
letters$isB=as.factor(letters$letter=='B')  
require('caTools')

## Loading required package: caTools

## Warning: package 'caTools' was built under R version 3.1.3

set.seed(1000)
spl=sample.split(letters$isB,SplitRatio=0.5)
train=subset(letters,spl==TRUE)
test=subset(letters,spl==FALSE)
# find the accurary of the baseline method in the test set
table(test$isB)

## 
## FALSE  TRUE 
##  1175   383

1175/(1175+383)

## [1] 0.754172

#build a classifiation tree 
require('rpart')

## Loading required package: rpart

CARTb=rpart(isB~.-letter,data=train,method='class')
predictions=predict(CARTb,newdata=test,type='class')
table(test$isB,predictions)

##        predictions
##         FALSE TRUE
##   FALSE  1118   57
##   TRUE     43  340
